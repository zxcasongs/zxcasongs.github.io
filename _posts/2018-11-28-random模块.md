字母加数字 随机验证码


```
def code(n =6):
    d = ''
    for i in range(n):
        num = random.randint(0, 9)
        S = chr(random.randint(65, 90))
        s = S.lower()

        STR = random.choice([s,S])

        x = random.randint(1,2)
        if x ==1:
            d+=str(num)
        else:
            d+=str(STR)
    return d
print(code(4))
```
